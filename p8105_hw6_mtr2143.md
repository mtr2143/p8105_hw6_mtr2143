birthweight_df <-
  birthweight_data %>% 
  mutate(
    babysex = ifelse(babysex == 1, "male", "female") %>% 
      forcats::as_factor(),
    frace = case_when(
      frace == 1 ~ "white", 
      frace == 2 ~ "black", 
      frace == 3 ~ "asian", 
      frace == 4 ~ "puerto rican", 
      frace == 8 ~ "other", 
      frace == 9 ~ "unknown"
    ) %>% 
      forcats::as_factor(),
    malform = ifelse(malform == 0, "absent", "present") %>% 
      forcats::as_factor(),
